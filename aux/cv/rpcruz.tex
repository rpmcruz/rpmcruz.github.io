\caixaaberta{cv11}2021(june?), \textbf{PhD in Computer Science [waiting for defense]}\\co-joint degree between the University of Porto, Minho and Aveiro
\begin{adjustwidth}{1.5em}{}\color{black!70}\footnotesize The PhD thesis was on machine learning -- more specifically computer vision (Rethinking a Deep Learning Pipeline for Images). The major highlights from the thesis:

\end{adjustwidth}
%
\begin{adjustwidth}{1.5em}{}%
%
\begin{minipage}[t]{0.88\linewidth}
\caixafechada{cv111}\color{black!70}\footnotesize\textbf{Background invariant training applied to electrical insulator detection:} Deep learning methods are vulnerable to changes in background, even if they do not affect the object of interest. We show that adversarial training (similar to GANs) can be used to greatly ameliorate this problem. The method was applied to electrical insulator detection, in collaboration with a team from Brazil.

\vspace{2pt}\end{minipage}\hfill\begin{minipage}[t]{0.1\linewidth}\strut\vspace*{-\baselineskip}\newline\includegraphics[width=\linewidth]{imgs/cv-prates.jpg}\end{minipage}
%
%
\begin{minipage}[t]{\linewidth}
\caixafechada{cv112}\color{black!70}\footnotesize\textbf{Improving accuracy under class imbalance conditions using ranking:} Sometimes not all groups are well-represented in the data: consider patients with differing disease severity, or fraud and fault detection where the anomaly is rare, or racial under-representation. We have borrowed concepts from the ranking literature to improve on the classification accuracy of these under-represented cases.\vspace{2pt}\end{minipage}
%
%
\begin{minipage}[t]{\linewidth}
\caixafechada{cv113}\color{black!70}\footnotesize\textbf{Iterative inference:} Deep learning predictions are typically produced in a single-shot during the forward-pass. The proposed method allows for predictions to take place during the backward-pass and be, therefore, iterative. This allows incremental improvements of the decisions, including those generated by third-party machine learning models.\vspace{2pt}\end{minipage}
%
%
\end{adjustwidth}%
